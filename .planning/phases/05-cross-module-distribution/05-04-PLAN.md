    - initialize `autoUpdater.autoDownload = true`
    - on `checking-for-update`, send `{ state: 'checking' }`
    - on `update-available`, send `{ state: 'available', version: info.version }`
    - on `download-progress`, send `{ state: 'downloading', percent: progress.percent }`
    - on `update-not-available`, send `{ state: 'not-available' }`
    - on `update-downloaded`, send `{ state: 'downloaded', version: info.version }`
    - on error, send `{ state: 'error', message: error.message }`
    - `start()` calls `autoUpdater.checkForUpdates()`
    - `installUpdate()` calls `autoUpdater.quitAndInstall()`

    onUpdateStatus: (callback) => {

      ipcRenderer.on('shell:update-status', listener)
      return () => ipcRenderer.removeListener('shell:update-status', listener)
    }
    installUpdate: () => ipcRenderer.invoke('shell:installUpdate')
